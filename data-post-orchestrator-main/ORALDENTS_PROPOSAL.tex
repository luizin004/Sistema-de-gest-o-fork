\sectiontitle{Banco de Dados}

\subsection{Estrutura Principal}
\begin{lstlisting}[language=SQL, caption=Tabelas Principais]
-- Configurações de Disparos
CREATE TABLE disparos_config (
    id UUID PRIMARY KEY,
    tipo TEXT, -- aniversario, limpeza, clareamento, consulta
    mensagem_template TEXT,
    horario_disparo TIME,
    dias_antes INTEGER,
    ativo BOOLEAN,
    zapi_instance_id TEXT,
    zapi_token TEXT
);

-- Pacientes
CREATE TABLE clientes (
    id UUID PRIMARY KEY,
    nome TEXT,
    telefone TEXT,
    email TEXT,
    data_aniversario DATE,
    ultima_consulta DATE,
    proxima_limpeza DATE,
    status TEXT
);

-- Histórico de Disparos
CREATE TABLE disparos_historico (
    id UUID PRIMARY KEY,
    cliente_id UUID REFERENCES clientes(id),
    tipo_disparo TEXT,
    mensagem_enviada TEXT,
    data_envio TIMESTAMP,
    status TEXT, -- enviado, falhou, pendente
    resposta_cliente TEXT
);
\end{lstlisting}
